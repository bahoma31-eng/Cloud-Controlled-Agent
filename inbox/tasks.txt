#!/usr/bin/env python3
import subprocess
import os

def run_command(command):
    try:
        result = subprocess.run(command, shell=True, check=True,
                                stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
        return result.stdout.strip()
    except subprocess.CalledProcessError as e:
        return f"Error executing '{command}': {e.stderr.strip()}"

def collect_repository_metadata():
    print("Collecting repository metadata...\n")
    
    # Retrieve git status to validate repo connectivity and local status
    print("Git Status:")
    git_status = run_command("git status")
    print(git_status + "\n")
    
    # Retrieve commit history (oneline)
    print("Git Commit History (--oneline):")
    git_log = run_command("git log --oneline")
    print(git_log + "\n")
    
    # Retrieve branch list including remote branches
    print("Git Branch List (local and remote):")
    git_branches = run_command("git branch -a")
    print(git_branches + "\n")
    
    # Retrieve file structure starting from current directory
    print("File Structure:")
    for root, dirs, files in os.walk("."):
        # Calculate indent for visual structure representation
        level = root.replace(os.getcwd(), "").count(os.sep)
        indent = "    " * level
        print(f"{indent}{os.path.basename(root)}/")
        subindent = "    " * (level + 1)
        for file in files:
            print(f"{subindent}{file}")

if __name__ == '__main__':
    collect_repository_metadata()
