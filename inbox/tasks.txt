المهمة الأولى: فحص أداء المعالج والعمليات المستهلكة
COMMAND: Get-Process | Sort-Object CPU -Descending | Select-Object -First 10 -Property Name, CPU, WorkingSet | ConvertTo-Json
المهمة الثانية: تقرير مساحة التخزين وتحليل المجلدات الكبيرة
COMMAND: $disk = Get-PSDrive C; $report = "Free Space: " + [math]::Round($disk.Free/1GB, 2) + " GB`n"; $topFolders = Get-ChildItem -Path $env:USERPROFILE -Directory | ForEach-Object { $size = (Get-ChildItem -Path $_.FullName -Recurse -File -ErrorAction SilentlyContinue | Measure-Object -Property Length -Sum).Sum / 1MB; [PSCustomObject]@{FolderName=$_.Name; SizeMB=[math]::Round($size, 2)} } | Sort-Object SizeMB -Descending | Select-Object -First 5; $report + ($topFolders | Out-String)
