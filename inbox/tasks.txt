import os
from datetime import datetime

# Define the outbox directory path
outbox_dir = input("Please enter the path to the outbox directory: ")

# Check if the path exists
if not os.path.exists(outbox_dir):
    print("The specified path does not exist.")
else:
    # List all files in the outbox directory
    try:
        files = [os.path.join(outbox_dir, f) for f in os.listdir(outbox_dir) if os.path.isfile(os.path.join(outbox_dir, f))]
    except Exception as e:
        print(f"An error occurred while listing files: {e}")
        exit()

    # Sort files by their last modification time
    try:
        files.sort(key=lambda x: os.path.getmtime(x))
    except Exception as e:
        print(f"An error occurred while sorting files: {e}")
        exit()

    # Keep only the last two files and delete the rest
    if len(files) > 2:
        files_to_delete = files[:-2]
        for file in files_to_delete:
            try:
                os.remove(file)
                print(f"Deleted: {file}")
            except OSError as e:
                print(f"Error deleting {file}: {e}")
    else:
        print("Not enough files to delete.")
