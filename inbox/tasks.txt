#!/usr/bin/env python3
import os
import sys
from datetime import datetime

def collect_repository_info(repo_path):
    repo_info = {
        'total_files': 0,
        'total_dirs': 0,
        'files': []
    }
    for dirpath, dirnames, filenames in os.walk(repo_path):
        repo_info['total_dirs'] += len(dirnames)
        repo_info['total_files'] += len(filenames)
        for file in filenames:
            file_path = os.path.join(dirpath, file)
            try:
                mtime = os.path.getmtime(file_path)
                mtime_str = datetime.fromtimestamp(mtime).strftime('%Y-%m-%d %H:%M:%S')
                size = os.path.getsize(file_path)
            except Exception as e:
                mtime_str = "N/A"
                size = "N/A"
            repo_info['files'].append({
                'path': file_path,
                'modified': mtime_str,
                'size': size
            })
    return repo_info

def print_repository_info(info):
    print("Repository Analysis:")
    print(f"Total directories: {info['total_dirs']}")
    print(f"Total files: {info['total_files']}")
    print("\nFile Details:")
    for file_info in info['files']:
        print(f"{file_info['path']} - Last Modified: {file_info['modified']} - Size: {file_info['size']} bytes")

if __name__ == '__main__':
    repo_path = input("Enter the path to the repository: ").strip()
    if not os.path.isdir(repo_path):
        print("The provided path is not a valid directory.")
        sys.exit(1)
    info = collect_repository_info(repo_path)
    print_repository_info(info)
